// FILTER: arreglar sameAs URLs y openingHours en RankMath schema
add_filter('rank_math/json_ld', function ($data, $jsonld) {
    if (!is_array($data)) return $data;

    $walker = function (&$node) use (&$walker) {
        if (!is_array($node)) return;

        // Fix sameAs URLs (burjassot → valencia)
        if (isset($node['sameAs']) && is_array($node['sameAs'])) {
            foreach ($node['sameAs'] as &$url) {
                if (is_string($url) && strpos($url, 'masquemedicos.com/traumatologo_burjassot') !== false) {
                    $url = str_replace('traumatologo_burjassot', 'traumatologo_valencia', $url);
                }
                if (is_string($url) && strpos($url, 'doctor-antonio-vitaleta') !== false) {
                    $url = str_replace('doctor-antonio-vitaleta', 'doctor-antonio-vilatela', $url);
                }
            }
            unset($url);
        }

        // Fix openingHours: limit to L-V 09:00-20:00 if currently 7 days
        if (isset($node['openingHours']) && is_array($node['openingHours'])) {
            $node['openingHours'] = ['Mo-Fr 09:00-20:00'];
        }
        if (isset($node['openingHours']) && is_string($node['openingHours']) &&
            strpos($node['openingHours'], 'Saturday') !== false) {
            $node['openingHours'] = ['Mo-Fr 09:00-20:00'];
        }

        // Recurse
        foreach ($node as $k => &$v) {
            if (is_array($v)) $walker($v);
        }
        unset($v);
    };

    if (isset($data['@graph']) && is_array($data['@graph'])) {
        foreach ($data['@graph'] as &$entity) { $walker($entity); }
        unset($entity);
    } else {
        $walker($data);
    }

    return $data;
}, 100, 2);
