// ============ ONE-SHOT: añadir sección "Equipo médico" con cards a pillar 735 ============
add_action('admin_init', function () {
    if (!current_user_can('manage_options')) return;
    if (empty($_GET['_run_p735_noche']) || $_GET['_run_p735_noche'] !== '1') return;

    $page_id = 735;
    $log = [];
    $raw = get_post_meta($page_id, '_elementor_data', true);
    if (empty($raw)) { echo 'No _elementor_data'; exit; }
    $data = json_decode($raw, true);
    if (!is_array($data)) { echo 'JSON invalido'; exit; }

    // Marker idempotencia
    $marker = 'vlxlnoche';
    $exists = false;
    $check = function ($node) use (&$check, $marker, &$exists) {
        if (!is_array($node)) return;
        if (isset($node['id']) && $node['id'] === $marker) { $exists = true; return; }
        if (!empty($node['elements'])) foreach ($node['elements'] as $c) $check($c);
    };
    foreach ($data as $top) { $check($top); if ($exists) break; }

    if ($exists) {
        echo "marker '$marker' ya existe (skip). Schemas dinámicos sí están activos.";
        exit;
    }

    // CSS de la sección "Equipo médico Unidad de Columna"
    $css = '<style>'
        . '.vlxleq{background:#F5F7FA;border-radius:14px;padding:42px 28px;margin:32px 0}'

        . '.vlxleq .sub{text-align:center;color:#555;font-size:1.05em;margin-bottom:32px;max-width:760px;margin-left:auto;margin-right:auto}'
        . '.vlxleq-grid{display:grid;grid-template-columns:repeat(auto-fit,minmax(280px,1fr));gap:24px;max-width:900px;margin:0 auto}'

        . '.vlxleq-card:hover{transform:translateY(-4px)}'
        . '.vlxleq-card h3{color:#1B2A4A;margin:14px 0 6px;font-size:1.25em}'
        . '.vlxleq-card .role{color:#25D366;font-weight:600;font-size:0.95em;margin-bottom:14px}'
        . '.vlxleq-card .creds{color:#555;font-size:0.92em;line-height:1.55;margin-bottom:16px;text-align:left;padding-left:8px}'
        . '.vlxleq-card .creds li{margin-bottom:4px;list-style-position:inside}'
        . '.vlxleq-card a.lk{display:inline-block;color:#1B2A4A;font-weight:600;text-decoration:underline;font-size:0.95em}'
        . '.vlxleq-card a.lk:hover{color:#25D366}'
        . '.vlxleq-card.featured{border-top-color:#1B2A4A;background:linear-gradient(180deg,#fff 0%,#FFF8E7 100%)}'
        . '</style>';

        // CARD 1: Dr. Vilatela (featured)

        . '<h3>Dr. Antonio Vilatela</h3>'

        . '<div class="vlxleq-card">'
        . '<h3>Dr. Pablo Jover</h3>'
        . '<div class="role">Cofundador ICEM · Cirujano de Columna</div>'

        \Elementor\Plugin::$instance->files_manager->clear_cache();

    }
    if (has_action('litespeed_purge_all')) {
        do_action('litespeed_purge_all');
        $log[] = 'LiteSpeed purga disparada';
    }
    clean_post_cache($page_id);

    echo nl2br(esc_html(implode("\n", $log)));
    exit;
